# Wielojęzyczne prompty testowe NL → nlp2env → .env
#
# Czym jest ten plik?
#   Katalog 26 scenariuszy e2e: prompt użytkownika (różne języki) → LLM (Ollama)
#   → MCP nlp2env_set_email → plik .env → walidacja expect=...
#
# Uruchomienie:
#   export SMTP_PASSWORD=e2e-test-secret-42
#   ./e2e-multilang.sh
#   # lub: make examples-multilang
#
# Dokumentacja: README.md w tym katalogu (sekcja prompts-multilang.txt)
#
# Format bloku:
#   [prompt:id]
#   lang=pl|en|de|...
#   source=llm
#   nl=<prompt w danym języku>
#   expect=SMTP_HOST=...
#   expect=SMTP_USER=...
#   expect=SMTP_PORT=...

[prompt:pl-gmail]
lang=pl
source=llm
nl=Ustaw skrzynkę SMTP Gmail: serwer smtp.gmail.com, użytkownik jan@firma.pl, port 587
expect=SMTP_HOST=smtp.gmail.com
expect=SMTP_USER=jan@firma.pl
expect=SMTP_PORT=587
